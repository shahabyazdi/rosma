# Counter App

usage.counter.description

```jsx
import { useObserver } from 'rosma';

export default function Counter() {
  const { count, setCount } = useObserver(0);

  return (
    <button onClick={() => setCount(count + 1)}>
      <span>the count is: </span>
      {count}
    </button>
  );
}
```

```tsx
import { useObserver } from 'rosma';

type State = {
  count: number;
};

export default function Counter() {
  const { count, setCount } = useObserver<State>(0);

  return (
    <button onClick={() => setCount(count + 1)}>
      <span>the count is: </span>
      {count}
    </button>
  );
}
```
